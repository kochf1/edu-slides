%%%
%%% (Step 2) Create new Slide Templates
%%% Documention about available layouts (commands), blocks, colors,
%%% and fonts is available on GitHub or reading through edu-slides.sty
%%%

%% @template Describe your new template here
%% params: text{1}
\newcommand{\slideMyNewSlide}[1]{ 
    \begin{frame}[plain]
    \elementLogo
    %\elementTitle{true}{Title for This Slide}
    \layoutText{/box/half/top}{c}{c}{/large}{#1} %% Large Text on Top Box
    \layoutColor{/box/half/bottom}{fau-blue}
    %\layoutText{/box/half/bottom}{c}{c}{/body}{Text} %% Text on Bottom Box
    \elementBanner
    \end{frame}
}

%%%
%%% (Step 3) Create new Slide Decks
%%% Just call the Slide Template that you created above
%%%

\begin{document}
    \slideMyNewSlide{Text for the TOP Box}
\end{document}
