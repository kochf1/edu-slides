%% @template Conceptual Visualization
%% Creates a new conptual design
%% Example:
%%      \slideConceptual[title=slideConcept, font=big, one=72\%]{!text-1}}
%% params: conceptParams{1}, explain_1{2}
\newcommand{\slideConceptual}[2][]{ 
    \pgfkeys{/conceptParams/.cd, #1}
    \pgfkeys{/conceptParams/title/.get=\myTitle}%
    \pgfkeys{/conceptParams/type/.get=\myType}%
    \pgfkeys{/conceptParams/font/.get=\myFont}%
    \pgfkeys{/conceptParams/color/.get=\myColor}%

    \begin{frame}[plain]

    \IfStrEq{\myType}{colorful}{%
        \elementTitle{\myTitle}
        Do something here! \\
        #2
        \elementBanner
    }{
        Do something else here! \\
        #2
    }
    \end{frame}
}
